# D7.1 Progress Tracking - Function Breakdown

## Purpose
This document defines the implementation tasks for Smart Learn's Progress Tracking feature (D7.1) - comprehensive analytics and learning progress monitoring.

---

## Overview

While the previous features implement learning and review functionality, D7.1 adds the critical piece of **visibility** - tracking student progress, performance analytics, and learning patterns. This feature:

1. **Captures** all learning interactions and answers
2. **Analyzes** performance metrics and mastery levels
3. **Visualizes** progress over time with charts and trends
4. **Identifies** weak areas and provides recommendations

This creates a data-driven learning experience that helps students understand their growth and focus on areas that need improvement.

### 1. Comprehensive Data Collection

### Progress Tracker Class
```javascript
class ProgressTracker {
  constructor() {
    this.sessions = new Map();
    this.progressCache = new Map();
  }

  async recordAnswer(sessionId, questionId, answerData) {
    // Record answer submission
    const record = {
      id: `record_${Date.now()}`,
      sessionId,
      questionId,
      timestamp: new Date(),
      ...answerData
    };

    // Update session metrics
    this._updateSessionMetrics(sessionId, answerData);

    // Persist record
    await this._storeAnswer(record);

    // Update cached progress
    await this._updateProgressCache(sessionId);

    return record;
  }

  _updateSessionMetrics(sessionId, answerData) {
    const session = this.sessions.get(sessionId);
    if (!session) return;

    session.stats.totalAnswers++;
    session.stats.correctAnswers += answerData.isCorrect ? 1 : 0;
    session.stats.avgResponseTime = this._calculateAvgResponseTime(session, answerData);
  }

  async getProgress(sessionId) {
    const cacheKey = `progress_${sessionId}`;
    if (this.progressCache.has(cacheKey)) {
      return this.progressCache.get(cacheKey);
    }

    // Calculate comprehensive progress
    const progress = await this._calculateProgress(sessionId);

    this.progressCache.set(cacheKey, progress);
    return progress;
  }
}
```

### Per-Concept Mastery Calculation

**Mastery Algorithm:**
```javascript
function calculateMastery(conceptAnswers) {
  const total = conceptAnswers.length;
  const correct = conceptAnswers.filter(a => a.isCorrect).length;
  const firstAttempt = conceptAnswers.filter(a => a.attempt === 1).length;
  
  // Base accuracy score
  const accuracy = correct / total;
  
  // Bonus for quick correct answers
  const speedBonus = conceptAnswers
    .filter(a => a.isCorrect && a.responseTime < 30)
    .length / total;
  
  // Penalty for many retries
  const retryPenalty = conceptAnswers
    .filter(a => a.attempt > 1)
    .length / total;
  
  // Weighted calculation
  const mastery = (accuracy * 0.6) + (speedBonus * 0.3) - (retryPenalty * 0.1);
  
  return Math.round(Math.max(0, Math.min(100, mastery * 100)));
}
```

**Mastery Levels:**
- 0-20%: Novice (needs foundation)
- 21-40%: Emerging (building understanding)
- 41-60%: Developing (solidifying knowledge)
- 61-80%: Proficient (strong grasp)
- 81-95%: Advanced (mastery approaching)
- 96-100%: Expert (full mastery)

### Analytics Service Class
```javascript
class AnalyticsService {
  async calculateSessionProgress(sessionId) {
    const session = await this._getSession(sessionId);
    
    return {
      sessionId,
      duration: session.endTime - session.startTime,
      totalQuestions: session.stats.totalQuestions,
      answered: session.stats.answered,
      accuracy: session.stats.accuracy,
      avgResponseTime: session.stats.avgResponseTime,
      confidence: session.stats.confidenceRating
    };
  }

  async calculateLearningProgress(sessionId) {
    const allSessions = await this._getAllSessions(sessionId);
    
    return {
      totalSessions: allSessions.length,
      totalQuestions: allSessions.reduce((sum, s) => sum + s.stats.totalQuestions, 0),
      overallAccuracy: this._calculateOverallAccuracy(allSessions),
      masteryLevels: this._calculateMasteryLevels(allSessions),
      progressTrend: this._calculateProgressTrend(allSessions),
      weakAreas: this._identifyWeakAreas(allSessions),
      recommendations: this._generateRecommendations(allSessions)
    };
  }

  _identifyWeakAreas(allSessions) {
    // Find concepts with lowest mastery
    const conceptStats = this._aggregateConceptStats(allSessions);
    
    return Object.entries(conceptStats)
      .filter(([_, stats]) => stats.mastery < 50)
      .map(([conceptId, stats]) => ({
        conceptId,
        mastery: stats.mastery,
        errorRate: stats.errorRate,
        avgResponseTime: stats.avgResponseTime,
        priority: stats.mastery < 30 ? 'high' : 'medium'
      }))
      .sort((a, b) => a.mastery - b.mastery)
      .slice(0, 5); // Top 5 weakest
  }

  _generateRecommendations(allSessions) {
    const weakAreas = this._identifyWeakAreas(allSessions);
    const trends = this._calculateProgressTrend(allSessions);
    
    const recommendations = [];
    
    if (weakAreas.length > 0) {
      recommendations.push({
        type: 'focus',
        priority: 'high',
        message: `Focus on ${weakAreas[0].mastery}% mastered concepts`,
        targetConcepts: weakAreas.slice(0, 3)
      });
    }
    
    if (trends === 'declining') {
      recommendations.push({
        type: 'review',
        priority: 'medium',
        message: 'Performance trending downward. Consider review session.'
      });
    }
    
    if (trends === 'stable') {
      recommendations.push({
        type: 'advance',
        priority: 'low',
        message: 'Steady progress. Ready to advance to new topics.'
      });
    }
    
    return recommendations;
  }
}
```

### Database Schema (Recommended)
```sql
CREATE TABLE progress_records (
  id UUID PRIMARY KEY,
  session_id UUID,
  question_id UUID,
  concept_id UUID,
  is_correct BOOLEAN,
  is_partial BOOLEAN,
  attempt_number INT DEFAULT 1,
  response_time_seconds INT,
  confidence_rating INT,
  answer_text TEXT,
  created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE session_summaries (
  id UUID PRIMARY KEY,
  session_id UUID,
  project_id UUID,
  total_questions INT,
  questions_answered INT,
  questions_correct INT,
  accuracy_percentage DECIMAL(5,2),
  avg_response_time INT,
  total_duration_seconds INT,
  start_time TIMESTAMP,
  end_time TIMESTAMP,
  created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE concept_mastery (
  session_id UUID,
  concept_id UUID,
  mastery_percentage DECIMAL(5,2),
  questions_practiced INT,
  correct_answers INT,
  first_mastery_date TIMESTAMP,
  last_practice_date TIMESTAMP,
  PRIMARY KEY (session_id, concept_id)
);

CREATE TABLE progress_trends (
  session_id UUID,
  date DATE,
  accuracy DECIMAL(5,2),
  questions_practiced INT,
  avg_mastery DECIMAL(5,2),
  PRIMARY KEY (session_id, date)
);
```
